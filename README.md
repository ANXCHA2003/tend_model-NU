### 2. ‡∏ï‡∏±‡∏ß‡∏≠‡∏¢‡πà‡∏≤‡∏á‡πÇ‡∏Ñ‡πâ‡∏î‡πÉ‡∏ä‡πâ‡∏á‡∏≤‡∏ô

```dart
import 'package:tflite_flutter/tflite_flutter.dart';
import 'package:image/image.dart' as img;

class ProductClassifier {
  Interpreter? _interpreter;
  List<String>? _labels;

  Future<void> loadModel() async {
    // ‡πÇ‡∏´‡∏•‡∏î‡πÇ‡∏°‡πÄ‡∏î‡∏•
    _interpreter = await Interpreter.fromAsset('assets/product_classifier.tflite');
    
    // ‡πÇ‡∏´‡∏•‡∏î labels
    String labelData = await DefaultAssetBundle.of(context)
        .loadString('assets/class_names.json');
    _labels = json.decode(labelData).cast<String>();
  }

  Future<String> classifyImage(String imagePath) async {
    // ‡πÄ‡∏ï‡∏£‡∏µ‡∏¢‡∏°‡∏£‡∏π‡∏õ‡∏†‡∏≤‡∏û
    img.Image image = img.decodeImage(File(imagePath).readAsBytesSync())!;
    img.Image resized = img.copyResize(image, width: 224, height: 224);
    
    // ‡πÅ‡∏õ‡∏•‡∏á‡πÄ‡∏õ‡πá‡∏ô input tensor
    var input = imageToByteListFloat32(resized, 224, 224, 127.5, 127.5);
    
    // ‡∏ó‡∏≥‡∏ô‡∏≤‡∏¢
    var output = List.filled(1 * _labels!.length, 0.0).reshape([1, _labels!.length]);
    _interpreter!.run(input, output);
    
    // ‡∏´‡∏≤‡∏ú‡∏•‡∏•‡∏±‡∏û‡∏ò‡πå
    double maxScore = 0;
    int maxIndex = 0;
    for (int i = 0; i < output[0].length; i++) {
      if (output[0][i] > maxScore) {
        maxScore = output[0][i];
        maxIndex = i;
      }
    }
    
    return _labels![maxIndex];
  }
}
```
